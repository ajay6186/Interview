Examples 6.2 — Template and Slot

BASIC (1–13)

1. template element basics

<!-- The <template> element holds inert HTML that is never rendered. -->
<template id="my-tpl">
  <div class="card">
    <h2 class="title"></h2>
    <p  class="body"></p>
  </div>
</template>

<script>
  const tpl   = document.getElementById('my-tpl');
  const clone = tpl.content.cloneNode(true);

  clone.querySelector('.title').textContent = 'Hello';
  clone.querySelector('.body').textContent  = 'World';

  document.body.appendChild(clone);
</script>

The <template> element defines reusable inert markup. Its content is a DocumentFragment that can be cloned and inserted as many times as needed.

2. Template is inert HTML — scripts and images do not fire

<template id="lazy-tpl">
  <!-- ✅ Image src is NOT fetched while inside the template -->
  <img src="https://example.com/large-photo.jpg" alt="photo">

  <!-- ✅ Script is NOT executed while inside the template -->
  <script>console.log('This never runs while inert');</script>

  <!-- ✅ CSS :checked / :hover do not apply to template content -->
  <input type="checkbox" checked>
</template>

<!-- Only when cloneNode(true) result is inserted into the live DOM
     do images load, scripts execute, and styles apply. -->

Template content is completely inert: network requests are not made, scripts are not parsed, and CSS rules do not apply until the clone is inserted into the document.

3. template.content — accessing the DocumentFragment

<template id="row-tpl">
  <tr>
    <td class="name"></td>
    <td class="score"></td>
  </tr>
</template>

<script>
  const tpl = document.getElementById('row-tpl');

  // tpl.content is a DocumentFragment — a lightweight container node
  console.log(tpl.content.nodeType);     // 11 (DOCUMENT_FRAGMENT_NODE)
  console.log(tpl.content.childNodes);   // NodeList [tr]

  // querySelector works on the fragment:
  const td = tpl.content.querySelector('.name');
  console.log(td.tagName); // TD

  // The original template remains unchanged after querying
</script>

template.content returns the DocumentFragment containing the template's children. You can query it, clone it, and import it like any other DOM node.

4. cloneNode(true) — deep-cloning the template content

const tpl   = document.querySelector('#item-tpl');

// cloneNode(true) = deep clone — copies the entire subtree
const clone = tpl.content.cloneNode(true);
// clone is a DocumentFragment with all descendant elements copied

// Populate the clone before insertion (efficient — one reflow)
clone.querySelector('.label').textContent = 'Item A';
clone.querySelector('.value').textContent = '42';

document.querySelector('#list').appendChild(clone);

// ⚠️ After appendChild the clone's children are moved into the DOM.
// The clone DocumentFragment itself becomes empty.
// Always call cloneNode again for each additional copy.

cloneNode(true) performs a deep copy of the template's fragment. After append, the fragment is drained — clone again for each new instance.

5. appendChild clone — inserting template instances into the DOM

const tpl  = document.querySelector('#card-tpl');
const list = document.querySelector('#card-list');

const data = [
  { title: 'Alpha', desc: 'First item'  },
  { title: 'Beta',  desc: 'Second item' },
  { title: 'Gamma', desc: 'Third item'  },
];

data.forEach(item => {
  const clone = tpl.content.cloneNode(true);
  clone.querySelector('.title').textContent = item.title;
  clone.querySelector('.desc').textContent  = item.desc;
  list.appendChild(clone);  // Each clone inserts independently
});

// Result: three cards rendered from one template definition

Appending individual clones in a forEach loop is the standard pattern for rendering collections from a template — each clone is independent in the DOM.

6. querySelectorAll in template content

<template id="form-tpl">
  <label>Name: <input class="field" name="name"></label>
  <label>Age:  <input class="field" name="age" type="number"></label>
  <label>Email:<input class="field" name="email" type="email"></label>
</template>

<script>
  const tpl    = document.querySelector('#form-tpl');
  const fields = tpl.content.querySelectorAll('.field');
  // NodeList(3) [input, input, input]

  // Useful for introspecting a template before cloning,
  // e.g., to validate it has required slots:
  if (!fields.length) throw new Error('Template missing .field inputs');

  const clone = tpl.content.cloneNode(true);
  document.querySelector('form').appendChild(clone);
</script>

querySelectorAll on template.content lets you inspect, validate, or modify the template structure before cloning — the original remains unaffected.

7. Template not rendered until cloned — verify with the DOM

<template id="hidden-tpl">
  <p id="secret">Secret content</p>
</template>

<script>
  // The p#secret is NOT in the live document tree:
  console.log(document.getElementById('secret')); // null ← not there!

  // It lives only inside the template's content fragment:
  const tpl = document.getElementById('hidden-tpl');
  console.log(tpl.content.getElementById('secret')); // <p>

  // After insertion it becomes findable:
  document.body.appendChild(tpl.content.cloneNode(true));
  console.log(document.getElementById('secret')); // <p id="secret">
</script>

Template content is not part of the live DOM, so getElementById and querySelectorAll on document won't find it until a clone is inserted.

8. Multiple clones from one template

const tpl     = document.querySelector('#chip-tpl');
const wrapper = document.querySelector('.chips');
const tags    = ['HTML', 'CSS', 'JavaScript', 'WebComponents'];

// Each call to cloneNode produces an independent fragment
tags.forEach(tag => {
  const clone = tpl.content.cloneNode(true);
  clone.querySelector('.chip-label').textContent = tag;
  wrapper.appendChild(clone);
});

// All four chips are independent in the DOM.
// Changing one does not affect the others or the template.

A single template can be cloned unlimited times. The source template.content is never mutated — only the clones change when you populate them.

9. Template in head vs body — both work

<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <!-- ✅ Template in <head> is valid and works correctly -->
  <template id="head-tpl">
    <p>Defined in head</p>
  </template>
</head>
<body>
  <!-- ✅ Template in <body> is the most common placement -->
  <template id="body-tpl">
    <p>Defined in body</p>
  </template>

  <script>
    // Both are equally accessible:
    document.querySelector('#head-tpl').content.cloneNode(true);
    document.querySelector('#body-tpl').content.cloneNode(true);
  </script>
</body>
</html>

Templates can be placed in <head> or <body> — both are equally valid. Convention places them near the code that uses them, often at the bottom of body.

10. document.importNode — alternative to cloneNode

// document.importNode(node, deep) imports a node from another document
// or from a template content fragment.

const tpl   = document.querySelector('#row-tpl');
const clone = document.importNode(tpl.content, true);
// Equivalent to: tpl.content.cloneNode(true)
// But importNode is required when importing from a DIFFERENT document:

const iframe = document.querySelector('iframe');
const extTpl = iframe.contentDocument.querySelector('#ext-tpl');
// Must use importNode here — cloneNode would leave the node "owned"
// by the iframe document.
const imported = document.importNode(extTpl.content, true);
document.body.appendChild(imported);

document.importNode(template.content, true) is semantically equivalent to cloneNode(true) within the same document, but required when crossing document boundaries (e.g. iframes).

11. Template with <style>

<template id="styled-tpl">
  <style>
    /* These styles apply globally once the template is inserted! */
    /* This is a drawback — use shadow DOM to scope them. */
    .badge { display:inline-block; background:#6366f1; color:#fff;
             padding:2px 8px; border-radius:999px; font-size:12px; }
  </style>
  <span class="badge"></span>
</template>

<script>
  const clone = document.querySelector('#styled-tpl').content.cloneNode(true);
  clone.querySelector('.badge').textContent = 'New';
  document.body.appendChild(clone);
  // ⚠️ .badge styles are now GLOBAL — repeated clones add duplicate <style>.
  // Prefer shadow DOM + template combination to scope styles.
</script>

Styles in a non-shadow template become global when the clone is inserted. For scoped styles, combine the template with a shadow root via attachShadow().

12. Template with script — deferred execution

<template id="script-tpl">
  <div id="scripted-box"></div>
  <script>
    // This runs once, only when the clone is inserted into the document.
    document.getElementById('scripted-box').textContent = 'Script ran!';
  </script>
</template>

<script>
  // Script inside template does NOT run here:
  console.log(document.getElementById('scripted-box')); // null

  // Insert the clone — now the inner script executes:
  document.body.appendChild(
    document.querySelector('#script-tpl').content.cloneNode(true)
  );
  // Script ran, element is now in DOM.
  console.log(document.getElementById('scripted-box').textContent); // 'Script ran!'
</script>

Scripts inside templates execute when the clone enters the live DOM. Cloning the same template multiple times runs the script each time — be cautious of duplicates.

13. Template for table rows — clean separation of structure

<table id="scores-table">
  <thead><tr><th>Player</th><th>Score</th><th>Level</th></tr></thead>
  <tbody></tbody>
</table>

<template id="row-tpl">
  <tr>
    <td class="player"></td>
    <td class="score"></td>
    <td class="level"></td>
  </tr>
</template>

<script>
  const tbody = document.querySelector('#scores-table tbody');
  const tpl   = document.querySelector('#row-tpl');
  const rows  = [
    { player:'Alice', score:9800, level:5 },
    { player:'Bob',   score:7200, level:4 },
  ];
  rows.forEach(r => {
    const c = tpl.content.cloneNode(true);

    tbody.appendChild(c);
  });
</script>

Using a template for table rows keeps HTML structure out of JavaScript strings, making the row markup inspectable in DevTools and searchable in source code.

INTERMEDIATE (14–26)

14. Named slots in Shadow DOM

class ProfileCard extends HTMLElement {
  constructor() {
    super();
    this.attachShadow({ mode: 'open' }).innerHTML = `
      <style>
        :host   { display:flex; gap:12px; align-items:center; padding:12px;
                  border:1px solid #e5e7eb; border-radius:8px; }
        .info   { display:flex; flex-direction:column; gap:4px; }
        .name   { font-weight:600; }
        .detail { font-size:13px; color:#6b7280; }
      </style>
      <!-- Named slot for avatar -->
      <slot name="avatar"></slot>
      <div class="info">
        <span class="name"><slot name="name">Unknown</slot></span>
        <span class="detail"><slot name="role"></slot></span>
      </div>
    `;
  }
}
customElements.define('profile-card', ProfileCard);

// <profile-card>
//   <img slot="avatar" src="alice.jpg" width="48" height="48"
//        style="border-radius:50%">
//   <span slot="name">Alice Chen</span>
//   <span slot="role">Frontend Engineer</span>
// </profile-card>

Named slots route specific light DOM children to defined positions in the shadow DOM layout, letting consumers control content without touching the component's internal structure.

15. Default (unnamed) slot — catching all remaining children

class PaperBox extends HTMLElement {
  constructor() {
    super();
    this.attachShadow({ mode: 'open' }).innerHTML = `
      <style>
        :host { display:block; border:1px solid #e5e7eb; border-radius:8px;
                padding:16px; }
        header { font-weight:700; border-bottom:1px solid #e5e7eb; margin-bottom:12px; }
      </style>
      <header><slot name="header">Default Header</slot></header>
      <!-- Default slot catches ALL children without a slot= attribute -->
      <slot></slot>
    `;
  }
}
customElements.define('paper-box', PaperBox);

// <paper-box>
//   <span slot="header">My Box</span>
//   <p>Goes to default slot.</p>
//   <p>Also goes to default slot.</p>
// </paper-box>

The unnamed <slot> is the catch-all: any light DOM children that don't have a slot= attribute, or whose slot= name doesn't match, are projected here.

16. Slot fallback content

class NavItem extends HTMLElement {
  constructor() {
    super();
    this.attachShadow({ mode: 'open' }).innerHTML = `
      <style>
        :host { display:flex; align-items:center; gap:8px; padding:8px 12px;
                border-radius:6px; cursor:pointer; }
        :host(:hover) { background:#f3f4f6; }
      </style>
      <!-- Fallback shown when no icon is slotted -->
      <slot name="icon">
        <svg width="16" height="16" viewBox="0 0 24 24" fill="none"
             stroke="currentColor" stroke-width="2">
          <circle cx="12" cy="12" r="10"/>
        </svg>
      </slot>
      <!-- Fallback label -->
      <slot>Menu Item</slot>
    `;
  }
}
customElements.define('nav-item', NavItem);

// <nav-item>Dashboard</nav-item>           ← uses fallback icon + custom label
// <nav-item></nav-item>                     ← uses both fallbacks
// <nav-item><span slot="icon">🏠</span>Dashboard</nav-item> ← custom icon

Fallback content between <slot> tags renders only when no matching light DOM nodes are assigned to that slot — a convenient default state.

17. slotchange event — detecting slot content updates

class LiveList extends HTMLElement {
  constructor() {
    super();
    const shadow = this.attachShadow({ mode: 'open' });
    shadow.innerHTML = `
      <p id="count"></p>
      <slot></slot>
    `;

    // slotchange fires when nodes are added or removed from the slot
    shadow.querySelector('slot').addEventListener('slotchange', (e) => {
      const assigned = e.target.assignedElements();
      shadow.querySelector('#count').textContent =
        `${assigned.length} item(s) slotted`;
      console.log('Slotted elements:', assigned);
    });
  }
}
customElements.define('live-list', LiveList);

// Adding or removing children of <live-list> fires slotchange.

The slotchange event fires on the <slot> element whenever its assigned nodes change — useful for components that need to react to dynamic content injection.

18. assignedNodes() — reading all assigned nodes

class InspectSlot extends HTMLElement {
  constructor() {
    super();
    const shadow = this.attachShadow({ mode: 'open' });
    shadow.innerHTML = `<slot></slot>`;

    shadow.querySelector('slot').addEventListener('slotchange', (e) => {
      const slot = e.target;

      // assignedNodes() — returns Text nodes AND elements
      const all = slot.assignedNodes();
      console.log('All nodes (incl. text):', all);

      // assignedNodes({ flatten: true }) — resolves nested slots
      const flat = slot.assignedNodes({ flatten: true });
      console.log('Flattened:', flat);

      // assignedElements() — elements only (no text nodes)
      const els = slot.assignedElements();
      console.log('Elements only:', els);
    });
  }
}
customElements.define('inspect-slot', InspectSlot);

assignedNodes() returns all nodes (including text) distributed into a slot. assignedElements() filters to only element nodes. { flatten: true } resolves chained slots.

19. CSS ::slotted() — styling projected light DOM elements

class TagList extends HTMLElement {
  constructor() {
    super();
    this.attachShadow({ mode: 'open' }).innerHTML = `
      <style>
        :host { display:flex; flex-wrap:wrap; gap:6px; }

        /* Styles ALL slotted elements */
        ::slotted(*) {
          display:inline-block;
          padding:2px 10px;
          border-radius:999px;
          font-size:13px;
        }

        /* Target only slotted <span> elements */
        ::slotted(span) {
          background:#e0e7ff;
          color:#3730a3;
        }

        /* ⚠️ ::slotted only matches direct slotted children,
           NOT their descendants */
        ::slotted(span strong) { /* this does NOT work */ }
      </style>
      <slot></slot>
    `;
  }
}
customElements.define('tag-list', TagList);

::slotted(selector) styles elements from the light DOM that are projected into the shadow slot. It only matches direct children — not deeper descendants of slotted elements.

20. Projecting multiple elements into one slot

class ActionBar extends HTMLElement {
  constructor() {
    super();
    this.attachShadow({ mode: 'open' }).innerHTML = `
      <style>
        :host { display:flex; align-items:center; justify-content:space-between;
                padding:8px 16px; background:#f9fafb; border-radius:6px; }
        .left  { display:flex; gap:8px; }
        .right { display:flex; gap:8px; }
      </style>
      <div class="left"><slot name="start"></slot></div>
      <div class="right"><slot name="end"></slot></div>
    `;
  }
}
customElements.define('action-bar', ActionBar);

// Multiple elements with the same slot name all go to that slot:
// <action-bar>
//   <button slot="start">New</button>
//   <button slot="start">Import</button>
//   <button slot="end">Export</button>
//   <button slot="end">Settings</button>
// </action-bar>

Multiple light DOM elements with the same slot= value are all projected into the same slot and rendered in DOM order — slots are not limited to one element.

21. Template composition patterns — combining templates

// Define small focused templates, compose them into larger structures.

<template id="icon-tpl">
  <svg width="16" height="16" viewBox="0 0 24 24" fill="none"
       stroke="currentColor" stroke-width="2">
    <path class="icon-path"></path>
  </svg>
</template>

<template id="btn-tpl">
  <button class="icon-btn">
    <!-- Placeholder for composed icon -->
    <span class="icon-slot"></span>
    <span class="btn-label"></span>
  </button>
</template>

<script>
  function makeIconButton(iconPath, label) {
    const icon = document.querySelector('#icon-tpl').content.cloneNode(true);
    icon.querySelector('.icon-path').setAttribute('d', iconPath);

    const btn = document.querySelector('#btn-tpl').content.cloneNode(true);
    btn.querySelector('.icon-slot').appendChild(icon);
    btn.querySelector('.btn-label').textContent = label;
    return btn;
  }
  document.body.appendChild(makeIconButton('M12 4v16M4 12h16', 'Add'));
</script>

Template composition builds complex UI by cloning smaller templates and appending them into designated slots within a parent template — all without frameworks.

22. Template for modal dialog

<template id="modal-tpl">
  <div class="modal-backdrop">
    <div class="modal" role="dialog" aria-modal="true">
      <header class="modal-header">
        <h2 class="modal-title"></h2>
        <button class="modal-close" aria-label="Close">✕</button>
      </header>
      <div class="modal-body"></div>
      <footer class="modal-footer"></footer>
    </div>
  </div>
</template>

<script>
  function openModal({ title, body, footer }) {
    const clone = document.querySelector('#modal-tpl').content.cloneNode(true);
    clone.querySelector('.modal-title').textContent = title;
    clone.querySelector('.modal-body').innerHTML    = body;
    clone.querySelector('.modal-footer').innerHTML  = footer;
    clone.querySelector('.modal-close').addEventListener('click', () =>
      document.querySelector('.modal-backdrop').remove());
    document.body.appendChild(clone);
  }
  openModal({ title:'Confirm', body:'<p>Are you sure?</p>',
    footer:'<button>Yes</button> <button>No</button>' });
</script>

A modal template separates the dialog structure from its content. The factory function populates title, body, and footer from parameters before inserting.

24. Template for list item — efficient batch rendering

<template id="li-tpl">
  <li class="todo-item">
    <input type="checkbox" class="todo-check">
    <label class="todo-label"></label>
    <button class="todo-del" aria-label="Delete">✕</button>
  </li>
</template>

<script>
  // Use DocumentFragment to batch-insert all items in one reflow
  const tpl  = document.querySelector('#li-tpl');
  const frag = document.createDocumentFragment();
  const todos = ['Buy milk', 'Write tests', 'Deploy app'];

  todos.forEach((text, i) => {
    const c = tpl.content.cloneNode(true);
    const id = `todo-${i}`;
    c.querySelector('.todo-check').id  = id;
    c.querySelector('.todo-label').htmlFor     = id;
    c.querySelector('.todo-label').textContent = text;
    frag.appendChild(c);
  });

  document.querySelector('#todo-list').appendChild(frag);
  // Single DOM mutation for all items
</script>

Accumulating clones into a DocumentFragment before a single appendChild batches DOM mutations into one reflow — a significant performance win for long lists.

25. Template for notification toast

<template id="toast-tpl">
  <div class="toast" role="status" aria-live="polite">
    <span class="toast-icon"></span>
    <span class="toast-msg"></span>
    <button class="toast-close" aria-label="Dismiss">✕</button>
  </div>
</template>

<script>
  const ICONS = { success:'✅', error:'❌', warning:'⚠️', info:'ℹ️' };

  function showToast(msg, type = 'info', duration = 4000) {
    const c = document.querySelector('#toast-tpl').content.cloneNode(true);
    const el = c.querySelector('.toast');
    el.classList.add(`toast--${type}`);
    c.querySelector('.toast-icon').textContent = ICONS[type];
    c.querySelector('.toast-msg').textContent  = msg;
    c.querySelector('.toast-close').addEventListener('click', () => el.remove());
    document.querySelector('#toast-container').appendChild(c);
    setTimeout(() => el.remove(), duration);
  }

  showToast('File saved successfully', 'success');
</script>

A toast template produces accessible live-region notifications. The factory sets icon, message, and type modifier class, then auto-removes after the duration elapses.

39. Template performance vs innerHTML — when to use which

// ── innerHTML approach ────────────────────────────────────────────
// Simple, synchronous, but re-parses the HTML string on every call.
function renderViaInnerHTML(data) {
  container.innerHTML = data.map(item =>
    `<li class="item"><span>${item.name}</span></li>`
  ).join('');
}

// ── template approach ─────────────────────────────────────────────
// Template is parsed ONCE by the browser at page load.
// Subsequent clones are a native C++ tree copy — no re-parsing.
const tpl = document.querySelector('#item-tpl');
function renderViaTemplate(data) {
  const frag = document.createDocumentFragment();
  data.forEach(item => {
    const c = tpl.content.cloneNode(true);
    c.querySelector('span').textContent = item.name; // XSS-safe!
    frag.appendChild(c);
  });
  container.replaceChildren(frag); // single reflow
}

// Template wins for:
//  1. Large lists (no repeated HTML parsing)
//  2. Security (textContent assignment never injects script)
//  3. Readability (structure in HTML, data in JS)

Templates avoid repeated HTML parsing and enable safe textContent assignment — preventing XSS. For lists of 100+ items the difference in parse overhead becomes measurable.

43. Streaming template rendering with ReadableStream

// Stream large lists to avoid blocking the main thread.
// Uses scheduler.yield() (Chrome 115+) as a cooperative yield point.

const tpl = document.querySelector('#row-tpl');

async function streamRender(items, container) {
  const frag = document.createDocumentFragment();
  let batchSize = 50;

  for (let i = 0; i < items.length; i++) {
    const c = tpl.content.cloneNode(true);
    c.querySelector('.row-cell').textContent = items[i].name;
    frag.appendChild(c);

    // Yield to browser every batchSize items to stay responsive
    if (i % batchSize === batchSize - 1) {
      container.appendChild(frag);
      // scheduler.yield() suspends this task and resumes later
      if ('scheduler' in window) await scheduler.yield();
    }
  }
  container.appendChild(frag); // remainder
}

45. Template and virtual DOM comparison

// ── <template> element (native, imperative) ───────────────────────
// + No library; works everywhere HTML works
// + Memory-efficient: clone is a C++ operation
// - No declarative bindings; you set properties manually
// - No diffing — you replace or update nodes by hand

// ── Virtual DOM (React, Preact, etc.) ─────────────────────────────
// + Declarative JSX syntax
// + Automatic diffing — framework updates only changed DOM nodes
// - Extra JS bundle (~40 KB Preact, ~140 KB React)
// - Mental model mismatch with browser's real DOM

// ── Lit (thin layer over native web components) ───────────────────
// + Declarative tagged-template syntax
// + Efficient part-based updates (no full VDOM diff)
// + Ships as native custom elements (interoperable)
// - Still a dependency (~5 KB min+gz)

// Rule of thumb:
// Static or rarely-updated lists → <template> + cloneNode
// Highly dynamic, complex state   → framework or Lit
// Design system components        → custom elements + shadow DOM

The native template element, virtual DOM frameworks, and Lit each make different tradeoffs — native templates excel at zero-dependency simplicity; VDOM frameworks at rich reactive UIs.

47. Template caching — reuse across modules

// templates.js — centralised template registry
const cache = new Map();

export function getTemplate(id) {
  if (cache.has(id)) return cache.get(id);

  const tpl = document.getElementById(id);
  if (!tpl) throw new Error(`Template #${id} not found`);

  // Cache the content fragment (not the template element itself)
  // so we can always call cloneNode on the same reference.
  cache.set(id, tpl.content);
  return tpl.content;
}

export function clone(id) {
  return getTemplate(id).cloneNode(true);
}

A template registry centralises lookup and caches the content reference, preventing repeated getElementById calls across modules while keeping the clone pattern consistent.

48. Accessible focus management after template insertion

<template id="dialog-tpl">
  <div class="modal" role="dialog" aria-modal="true">
    <button class="modal-close">Close</button>
    <h2 class="modal-title" tabindex="-1"></h2>
    <div class="modal-body"></div>
  </div>
</template>

<script>
let lastFocused;

function openDialog(title, bodyHTML) {
  lastFocused = document.activeElement; // remember trigger for return focus

  const c = document.querySelector('#dialog-tpl').content.cloneNode(true);
  c.querySelector('.modal-title').textContent = title;
  c.querySelector('.modal-body').innerHTML    = bodyHTML;
  document.body.appendChild(c);

  const modal = document.querySelector('.modal');
  // Move focus into the dialog immediately for keyboard / SR users
  modal.querySelector('.modal-title').focus();

  modal.querySelector('.modal-close').addEventListener('click', closeDialog);
}

function closeDialog() {
  document.querySelector('.modal')?.remove();
  lastFocused?.focus(); // return focus to trigger element
}
</script>

After inserting a dialog template, focus must move into the dialog and return to the trigger on close — required by WCAG 2.5.3 and the ARIA dialog pattern for keyboard and screen reader users.

49. Template with CSS custom highlight API

<template id="search-result-tpl">
  <li class="result">
    <a class="result-link" href="">
      <strong class="result-title"></strong>
    </a>
    <p class="result-snippet"></p>
  </li>
</template>

<script>
// CSS Custom Highlight API (Chrome 105+) marks text ranges
// without wrapping them in spans — ideal for search results.
function renderResults(results, query) {
  const tpl  = document.querySelector('#search-result-tpl');
  const list = document.querySelector('#results');
  const ranges = [];
  list.innerHTML = '';

  results.forEach(({ title, url, snippet }) => {
    const c = tpl.content.cloneNode(true);
    c.querySelector('.result-link').href           = url;
    c.querySelector('.result-title').textContent   = title;
    c.querySelector('.result-snippet').textContent = snippet;
    list.appendChild(c);
  });

  // After insertion, highlight query terms with CSS Highlight API
  const re = new RegExp(query, 'gi');
  list.querySelectorAll('.result-snippet').forEach(el => {
    const text = el.firstChild;
    let match;
    while ((match = re.exec(el.textContent)) !== null) {
      const range = new Range();
      range.setStart(text, match.index);
      range.setEnd(text, match.index + match[0].length);
      ranges.push(range);
    }
  });
  CSS.highlights?.set('search-hit', new Highlight(...ranges));
}
// CSS: ::highlight(search-hit) { background: #fef08a; color: inherit; }
</script>

After rendering results from a template, the CSS Custom Highlight API marks search terms using Range objects without modifying the DOM — giving highlighting without extra spans.

50. Template content security — preventing XSS

const tpl = document.querySelector('#post-tpl');

// ✅ Safe: textContent treats string as plain text — no HTML parsed
function safeRender(post) {
  const c = tpl.content.cloneNode(true);
  c.querySelector('.title').textContent   = post.title;   // safe
  c.querySelector('.author').textContent  = post.author;  // safe
  c.querySelector('.body').textContent    = post.body;    // safe
  return c;
}

// ❌ Dangerous: innerHTML interprets <script> tags in user data
function unsafeRender(post) {
  const c = tpl.content.cloneNode(true);
  c.querySelector('.body').innerHTML = post.body; // ← XSS risk!
  return c;
}

// ✅ If you MUST render HTML: use the Sanitizer API (Chrome 105+)
// const sanitizer = new Sanitizer();
// el.setHTML(userContent, { sanitizer }); // sanitises before insertion

// ✅ Or use DOMPurify as a polyfill for older browsers:
// el.innerHTML = DOMPurify.sanitize(userContent);

Template-based rendering is inherently safer than string concatenation because textContent assignment never interprets HTML. When rich HTML is required, use the Sanitizer API or DOMPurify.
