Dialog & Details — 50 Examples

1. Basic <dialog> element

<dialog id="my-dialog">
  <p>Hello from a dialog!</p>
  <button onclick="this.closest('dialog').close()">Close</button>
</dialog>
<button onclick="document.getElementById('my-dialog').showModal()">Open</button>

2. showModal() vs show()

dialog.showModal(); // modal: adds ::backdrop, traps focus, blocks background
dialog.show();      // non-modal: no backdrop, no focus trap

3. close() method

dialog.close();           // close with no return value
dialog.close('cancelled'); // close and set returnValue
console.log(dialog.returnValue); // 'cancelled'

4. dialog open attribute

<dialog open>
  <!-- visible immediately, non-modal -->
</dialog>

<!-- Prefer showModal()/show() over the open attribute -->

5. Detecting when dialog closes

dialog.addEventListener('close', () => {
  console.log('Dialog closed with:', dialog.returnValue);
});

6. Detecting dialog cancel (Escape key)

dialog.addEventListener('cancel', e => {
  e.preventDefault(); // prevent default close if needed
  console.log('User pressed Escape');
});

7. Closing dialog with backdrop click

dialog.addEventListener('click', e => {
  if (e.target === dialog) { // click on backdrop (dialog element itself)
    dialog.close();
  }
});

37. Dialog vs details — when to use which

// Use <dialog> when:
// - Content requires user action/confirmation
// - Content should block background interaction
// - A separate "window" feel is needed

// Use <details> when:
// - Content is supplementary/optional
// - Inline toggle within page flow
// - No action required from user

44. Confirmation pattern using returnValue

async function confirm(message) {
  const dialog = document.createElement('dialog');
  dialog.innerHTML = `
    <p>${message}</p>
    <form method="dialog">
      <button value="cancel">Cancel</button>
      <button value="ok">OK</button>
    </form>`;
  document.body.appendChild(dialog);
  dialog.showModal();
  return new Promise(resolve => {
    dialog.addEventListener('close', () => {
      resolve(dialog.returnValue === 'ok');
      dialog.remove();
    });
  });
}

// Usage
if (await confirm('Delete this file?')) { deleteFile(); }
