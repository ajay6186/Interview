Data Attributes — 50 Examples

1. Basic data attribute

<div data-user-id="42">John Doe</div>

2. Reading data attribute with getAttribute

const el = document.querySelector('[data-user-id]');
const id = el.getAttribute('data-user-id'); // "42"

3. Reading via dataset API

const el = document.querySelector('[data-user-id]');
const id = el.dataset.userId; // "42" — note camelCase conversion

4. HTML to dataset camelCase conversion

data-user-id        → dataset.userId
data-first-name     → dataset.firstName
data-max-value      → dataset.maxValue
data-my-long-attr   → dataset.myLongAttr

5. Setting data attribute via dataset

el.dataset.status = 'active'; // sets data-status="active"
el.dataset.userId = 99;       // sets data-user-id="99" (converts to string)

6. Setting data attribute via setAttribute

el.setAttribute('data-theme', 'dark');
// Equivalent to el.dataset.theme = 'dark';

7. Removing a data attribute

delete el.dataset.status;          // removes data-status
el.removeAttribute('data-status'); // equivalent

8. Checking for a data attribute

if ('userId' in el.dataset) {
  console.log('data-user-id exists');
}
// or
if (el.hasAttribute('data-user-id')) { ... }

9. Iterating all data attributes

Object.entries(el.dataset).forEach(([key, value]) => {
  console.log(key, value); // camelCase keys
});

10. CSS attribute selector with data attribute

[data-theme="dark"] {
  background: #1a1a1a;
  color: #eee;
}
[data-status="active"] {
  border-left: 4px solid green;
}

13. CSS substring selector with data attribute

[data-category*="sport"] { border-color: green; }
[data-category^="news"]  { border-color: blue; }
[data-category$="live"]  { border-color: red; }

14. Storing configuration on a widget root

<div id="chart"
     data-type="bar"
     data-width="600"
     data-animate="true">
</div>

<script>
  const chart = document.getElementById('chart');
  initChart({
    type:    chart.dataset.type,
    width:   Number(chart.dataset.width),
    animate: chart.dataset.animate === 'true',
  });
</script>

15. Data attributes for filtering (CSS + JS)

<ul>
  <li data-category="fruit">Apple</li>
  <li data-category="veggie">Broccoli</li>
  <li data-category="fruit">Banana</li>
</ul>

<script>
  function filter(cat) {
    document.querySelectorAll('li').forEach(li => {
      li.style.display = li.dataset.category === cat ? '' : 'none';
    });
  }
</script>

16. Data attribute for sort order

<tr data-sort-key="2026-01-15">...</tr>

<script>
  const rows = [...tbody.querySelectorAll('tr')];
  rows.sort((a, b) => a.dataset.sortKey.localeCompare(b.dataset.sortKey));
  rows.forEach(row => tbody.appendChild(row));
</script>

17. Data attribute for tooltip content

<span data-tooltip="This is a tooltip">Hover me</span>

<script>
  document.querySelectorAll('[data-tooltip]').forEach(el => {
    el.addEventListener('mouseenter', () => showTooltip(el.dataset.tooltip));

18. Data attributes vs hidden input for form data

<!-- data attribute (not submitted with form) -->
<div data-user-id="42">...</div>

<!-- hidden input (submitted with form) -->
<input type="hidden" name="userId" value="42">

19. Delegated event handling via data attributes

<div id="actions">
  <button data-action="save">Save</button>
  <button data-action="delete">Delete</button>
  <button data-action="print">Print</button>
</div>

<script>
  document.getElementById('actions').addEventListener('click', e => {
    const action = e.target.closest('[data-action]')?.dataset.action;
    if (action) handlers[action]?.();
  });
</script>

20. Storing JSON in a data attribute

<div data-config='{"color":"blue","size":12}'></div>

<script>
  const config = JSON.parse(el.dataset.config);
  console.log(config.color); // "blue"
</script>
Use single quotes for the attribute and double quotes inside JSON.

21. dataset values are always strings

<div data-count="5" data-active="true"></div>

<script>
  typeof el.dataset.count  // "string" — NOT number
  typeof el.dataset.active // "string" — NOT boolean

  // Convert explicitly:
  const count  = Number(el.dataset.count);   // 5
  const active = el.dataset.active === 'true'; // true
</script>

23. Data attribute as state machine

<button data-state="idle">Submit</button>

<script>
  const states = { idle: 'Submit', loading: 'Saving...', success: 'Saved!' };
  function setState(btn, state) {
    btn.dataset.state = state;
    btn.textContent = states[state];
  }
  submitBtn.addEventListener('click', async () => {
    setState(submitBtn, 'loading');
    await save();
    setState(submitBtn, 'success');
  });
</script>

27. Data attribute for lazy loading trigger

<img data-src="real-image.jpg" src="placeholder.svg" alt="Product">

<script>
  const observer = new IntersectionObserver(entries => {
    entries.forEach(e => {
      if (e.isIntersecting) {
        e.target.src = e.target.dataset.src;
        observer.unobserve(e.target);
      }
    });
  });
  document.querySelectorAll('[data-src]').forEach(img => observer.observe(img));
</script>

29. Data attribute for step wizard

<div data-step="1" data-total-steps="4">
  Step 1 of 4
</div>

<script>
  function nextStep(el) {
    const current = Number(el.dataset.step);
    const total   = Number(el.dataset.totalSteps);
    if (current < total) el.dataset.step = current + 1;
  }
</script>

34. Data attribute for game state

<div id="board" data-turn="X" data-moves="0"></div>

<script>
  function makeMove(cell) {
    const board = document.getElementById('board');
    cell.dataset.mark = board.dataset.turn;
    board.dataset.turn = board.dataset.turn === 'X' ? 'O' : 'X';
    board.dataset.moves = Number(board.dataset.moves) + 1;
  }
</script>

40. Mutation observer watching data attribute changes

const observer = new MutationObserver(mutations => {
  mutations.forEach(m => {
    if (m.attributeName === 'data-status') {
      console.log('Status changed to:', m.target.dataset.status);
    }
  });
});
observer.observe(el, { attributes: true, attributeFilter: ['data-status'] });

44. Performance: dataset vs getAttribute

// dataset — slightly slower due to DOMStringMap proxy; fine for most use cases
const val1 = el.dataset.userId;

// getAttribute — slightly faster raw access
const val2 = el.getAttribute('data-user-id');

// Difference is negligible in practice; use dataset for readability

50. Security — never trust data attribute values from users

<!-- data-* values are user-visible/editable in DevTools -->
<!-- NEVER use them for auth/access control in the DOM -->

// BAD — client can change data-role in DevTools
if (el.dataset.role === 'admin') grantAccess(); // don't do this

// GOOD — check role server-side or from a signed token
